      const int buttonPin = 2;     // button is connected to digital pin 2
      const int yellowLedPin = 9;  // yellow LED is connected to digital pin 9
      const int greenLedPin = 10;  // green LED is connected to digital pin 10
      const int redLedPin = 11;    // red Led is connected to digital pin 11
      
      int buttonState = 0; // the buttons status
      
      void setup() {
        pinMode(yellowLedPin, OUTPUT);  //setting the yellow LED as an output
        pinMode(greenLedPin, OUTPUT);   //setting the green LED as an ouput
        pinMode(redLedPin, OUTPUT);     //setting the red LED as an output
        pinMode(buttonPin, INPUT);     //setting the buttonPin as an input
      
      }
      
      void loop() {
        // reads the state of the button, if it has been pressed(on or off)
        buttonState = digitalRead(buttonPin);
        // if the button is pressed, it is represented as high
        if(buttonState == HIGH){
        // turns yellow Led on
          digitalWrite(yellowLedPin, HIGH);
        // turns green Led on
          digitalWrite(greenLedPin, HIGH);
        // turns red Led on
          digitalWrite(redLedPin, HIGH);
          
      
        // fade in from min to max in increments of 5 points
        // sets the value (range from 0 to 255)
        for (int fadeValue = 0; fadeValue <= 255; fadeValue += 5) {
          analogWrite(yellowLedPin, fadeValue);
          analogWrite(greenLedPin, fadeValue);
          analogWrite(redLedPin, fadeValue);
          // wait for 30 milliseconds to see the dimming effect
          delay(30);
        }
      
        // fade out from max to min in increments of 5 points:
        // sets the value (range from 0 to 255):
        for (int fadeValue = 255; fadeValue >= 0; fadeValue -= 5) {
          analogWrite(yellowLedPin, fadeValue);
          analogWrite(greenLedPin, fadeValue);
          analogWrite(redLedPin, fadeValue);
          // wait for 30 milliseconds to see the dimming effect
          delay(30);
        }
      } else {
      digitalWrite(yellowLedPin, LOW);
      digitalWrite(greenLedPin, LOW);
      digitalWrite(redLedPin, LOW);
      }
    }
